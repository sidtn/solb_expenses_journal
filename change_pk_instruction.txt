def create_uuid(apps, schema_editor):
    Category = apps.get_model("journal_api", "Category")
    objs = [obj for obj in Category.objects.filter(uuid=None)]
    with transaction.atomic():
        for obj in objs:
            obj.uuid = uuid.uuid4()
        Category.objects.bulk_update(objs, ["uuid"])
